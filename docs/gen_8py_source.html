60def next_point(pos: Point, angle: HexagonAngle, space: float) -> Point:
61 degrees = angle.value
62 angle_rad = toRads(degrees)
63 x = pos.x + space * cos(angle_rad)
64 y = pos.y + space * sin(angle_rad)
65 return Point(x, y)
